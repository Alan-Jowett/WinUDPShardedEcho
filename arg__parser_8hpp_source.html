69 void parse(int argc, const char* const argv[]) {
70 for (int i = 1; i < argc; ++i) {
71 const char* a = argv[i];
72 if (std::strncmp(a, "--", 2) == 0) {
73 const char* name = a + 2;
74 const char* eq = std::strchr(name, '=');
75 std::string key;
76 std::string val;
77 if (eq) {
78 key.assign(name, eq - name);
79 val = std::string(eq + 1);
80 } else {
81 key = name;
82 }
83 auto it = opts_.find(key);
84 if (it == opts_.end()) continue;
85 if (it->second.takes_value) {
86 if (!val.empty()) {
87 it->second.value = val;
88 it->second.set = true;
89 } else if (i + 1 < argc) {
90 it->second.value = argv[++i];
91 it->second.set = true;
92 }
93 } else {
94 // flag
95 it->second.value = "1";
96 it->second.set = true;
97 }
98 } else if (a[0] == '-' && a[1] != '\0') {
99 // short flags (only single-character supported)
100 char shortname = a[1];
101 auto itmap = short_to_long_.find(shortname);
102 if (itmap == short_to_long_.end()) continue;
103 auto it = opts_.find(itmap->second);
104 if (it == opts_.end()) continue;
105 if (it->second.takes_value) {
106 if (i + 1 < argc) {
107 it->second.value = argv[++i];
108 it->second.set = true;
109 }
110 } else {
111 it->second.value = "1";
112 it->second.set = true;
113 }
114 } else {
115 // Positional arguments are ignored by the parser
116 }
117 }
118 }

145 void print_help(const char* program_name) const {
146 std::cout << "Usage: " << program_name << " [options]\n";
147 std::cout << "Options:\n";
148 // Build display strings and compute column widths
149 struct Row {
150 std::string optnames;
151 std::string default_val;
152 std::string desc;
153 };
154 std::vector<Row> rows;
155 size_t opt_col_width = 0;
156 size_t def_col_width = 0;
157 for (const auto& kv : opts_) {
158 const std::string& long_name = kv.first;
159 const Option& opt = kv.second;
160 std::string names = "--" + long_name;
161 if (opt.short_name != '\0') {
162 names += ", -";
163 names.push_back(opt.short_name);
164 }
165 if (opt.takes_value) names += " <value>";
166 std::string def = opt.default_value;
167 if (def.empty()) def = "";
168 std::string desc = opt.description;
169 rows.push_back({names, def, desc});
170 opt_col_width = std::max(opt_col_width, names.size());
171 def_col_width = std::max(def_col_width, def.size());
172 }
173
174 // Print rows with aligned columns. Layout: two-space indent, optnames, two spaces,
175 // "Default:", pad, default, two spaces, description
176 for (const auto& r : rows) {
177 std::cout << " ";
178 std::cout << r.optnames;
179 // pad to opt_col_width
180 if (r.optnames.size() < opt_col_width)
181 std::cout << std::string(opt_col_width - r.optnames.size(), ' ');
182 std::cout << " ";
183 std::cout << "Default:";
184 // pad so that defaults align in a column
185 std::cout << " ";
186 if (r.default_val.size() < def_col_width)
187 std::cout << std::string(def_col_width - r.default_val.size(), ' ');
188 std::cout << r.default_val;
189 if (!r.desc.empty()) {
190 std::cout << " " << r.desc;
191 }
192 std::cout << "\n";
193 }
194 // Add custom help note for congestion controllers available
195 std::cout << "\nAvailable congestion controllers: null, bbr, reno\n";
196 }
