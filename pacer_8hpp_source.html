73 bool can_send() override {
74 if (unlimited_) return true;
75 // Allow congestion controller to influence effective rate
76 double target = cc_.target_rate_pps();
77 if (target > 0.0 && target != rate_pps_) {
78 rate_pps_ = target;
79 // recompute capacity when rate changes
80 capacity_ = (std::max)(1.0, rate_pps_ * 0.005);
81 }
82 uint64_t now = now_ns();
83 refill(now);
84 return tokens_ >= 1.0 - 1e-12;
85 }

93 void record_send(uint64_t seq) override {
94 if (unlimited_) return;
95 uint64_t now = now_ns();
96 refill(now);
97 // consume one token; do not allow negative debt
98 tokens_ = (std::max)(0.0, tokens_ - 1.0);

101 }

121 uint64_t get_next_send_time_ns() const override {
122 if (unlimited_) return 0;
123 uint64_t now = now_ns();
124 // compute tokens as if we refilled now (without mutating state)
125 double tokens_at_now = tokens_;
126 if (now > last_refill_ns_) {
127 double delta_s = static_cast<double>(now - last_refill_ns_) / 1e9;
128 tokens_at_now = (std::min)(capacity_, tokens_at_now + rate_pps_ * delta_s);
129 }
130 if (tokens_at_now >= 1.0) return 0;
131 double deficit = 1.0 - tokens_at_now; // tokens needed
132 double wait_s = deficit / rate_pps_;
133 uint64_t wait_ns = static_cast<uint64_t>(wait_s * 1e9);
134 return wait_ns;
135 }

164 void refill(uint64_t now_ns) {
165 if (now_ns <= last_refill_ns_) return;
166 double delta_s = static_cast<double>(now_ns - last_refill_ns_) / 1e9;
167 double add = rate_pps_ * delta_s;
168 tokens_ = (std::min)(capacity_, tokens_ + add);
169 last_refill_ns_ = now_ns;
