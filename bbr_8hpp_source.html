94 if (rtt_est_ns_ == 0)
95 rtt_est_ns_ = rtt_ns;

99 // Add this ACK timestamp to sliding window and evict old samples
100 ack_history_.push_back(now_ns);
101 const uint64_t window_ns = ack_window_ns_;
102 while (!ack_history_.empty() && ack_history_.front() + window_ns < now_ns)
103 ack_history_.pop_front();
104
105 // Compute sample delivered packets/sec over the window
106 double sample_pps = 0.0;
107 if (!ack_history_.empty()) {
108 double window_s =
109 static_cast<double>(std::max<uint64_t>(1, now_ns - ack_history_.front())) * 1e-9;
110 if (window_s > 0.0) sample_pps = static_cast<double>(ack_history_.size()) / window_s;
111 }
112
113 if (bandwidth_pps_ == 0.0)
114 bandwidth_pps_ = sample_pps;
115 else
116 bandwidth_pps_ = bandwidth_pps_ * 0.9 + sample_pps * 0.1;
117
118 // Compute new target but never exceed the initially requested rate
119 double new_target = std::max(bandwidth_pps_ * pacing_gain_, min_pps_);
120 if (initial_rate_pps_ > 0.0) new_target = std::min(new_target, initial_rate_pps_);
121 target_rate_.store(new_target);
122 }
