94 void on_ack(uint64_t now_ns, uint64_t seq, uint64_t rtt_ns) {
95 if (unlimited_) return;
96
97 // update min RTT observed and remember last RTT
98 if (rtt_ns > 0) {
99 if (min_rtt_ns_ == 0 || rtt_ns < min_rtt_ns_) min_rtt_ns_ = rtt_ns;
100 last_rtt_ns_ = rtt_ns;
101 }
102
103 // Remove send tracking
104 auto it = send_times_.find(seq);
105 if (it != send_times_.end()) send_times_.erase(it);
106
107 // Count ack and adjust cwnd according to Reno rules
108 if (cwnd_packets_ < ssthresh_) {
109 // slow start: increase by 1 packet per ACK (approx)
110 cwnd_packets_ += 1.0;
111 } else {
112 // congestion avoidance: increase by ~1 packet per RTT -> 1/cwnd per ACK
113 cwnd_packets_ += 1.0 / std::max(1.0, cwnd_packets_);
114 }
115
116 // clamp cwnd
117 if (cwnd_packets_ < 1.0) cwnd_packets_ = 1.0;
118 if (cwnd_packets_ > 1e6) cwnd_packets_ = 1e6;
119
120 update_target_rate();
121 }

132 void on_poll(uint64_t now_ns) {
133 if (unlimited_) return;
134 // detect RTT inflation: if last rtt sample > min_rtt * threshold, treat as congestion
135 if (last_rtt_ns_ > 0 && min_rtt_ns_ > 0) {
136 if (last_rtt_ns_ > min_rtt_ns_ * rtt_inflation_threshold_) {
137 // multiplicative decrease
138 ssthresh_ = std::max(2.0, cwnd_packets_ / 2.0);
139 cwnd_packets_ = ssthresh_;
140 update_target_rate();
141 }
142 }

162 void update_target_rate() {
163 if (min_rtt_ns_ == 0) {
164 // keep existing pacing rate if we don't yet have a min RTT
165 return;
166 }
167 double min_rtt_s = static_cast<double>(min_rtt_ns_) / 1e9;
168 if (min_rtt_s <= 0.0) return;
169 pacing_rate_pps_ = cwnd_packets_ / min_rtt_s;
170 // Clamp to avoid runaway: limit to a multiplier of initial requested rate
171 const double max_mult = 1.0; // do not exceed initial requested rate
172 if (initial_rate_pps_ > 0.0) {
173 double cap = initial_rate_pps_ * max_mult;
174 if (pacing_rate_pps_ > cap) pacing_rate_pps_ = cap;
175 }
176 if (pacing_rate_pps_ < 1.0) pacing_rate_pps_ = 1.0;
177 }
