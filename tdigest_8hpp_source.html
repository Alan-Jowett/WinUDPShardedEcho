185 double cumulative = 0.0;

187 double current_weight = merged[0].second;

189 for (size_t i = 1; i < merged.size(); ++i) {

193 (void)cumulative;

196
197 if (current_weight + w <= (std::max)(1.0, k_limit)) {
198 // merge into current centroid
199 current_mean = (current_mean * current_weight + v * w) / (current_weight + w);
